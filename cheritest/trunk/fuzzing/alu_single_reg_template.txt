# Template for single instruction alu fuzz tests. The opcode and value
# to test are chosen in a script. Currently only R-type intructions are supported...

$params_str

.set mips64
.set noreorder
.set nobopt
.set noat

.global test
test:   .ent    test
	dli     $$a0, $a0_val
	dli     $$a1, $a1_val

	# Try different combinations of the two source
 	# registers. This doesn't actually add much but
        # doesn't cost anything either.
	$op0	$$a2, $$a0, $$a0
	$op0	$$a3, $$a0, $$a1
	$op0	$$a4, $$a1, $$a0
	$op0	$$a5, $$a1, $$a1

	jr	$$ra
	nop
        .end    test
